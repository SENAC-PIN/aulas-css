<style>
/* GUIA DA ATIVIDADE: Adicione estilo para input:focus e input[type='email']. Comece pela declaração padding indicada abaixo. Salve e compare o antes e o depois. */

.formulario input { padding: 10px; /* altere esta declaração e observe o efeito */ border: 1px solid #9fb0c1; }

.formulario button { padding: 10px; background: #355070; color: white; border: 0; }
</style>
