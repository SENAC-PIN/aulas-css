Código base

/* GUIA DA ATIVIDADE: Crie um mixin SASS com parâmetro, use @include e escreva o CSS que seria gerado. Localize no exemplo a parte relacionada ao tema e altere um valor por vez. Salve e compare o antes e o depois. */
// componentes.scss
@mixin botao($cor) {
  padding: 12px 18px;
  background: $cor;
}
.botao-ok { @include botao(#2a9d8f); }

/* CSS compilado */
.botao-ok {
  padding: 12px 18px;
  background: #2a9d8f;
}
